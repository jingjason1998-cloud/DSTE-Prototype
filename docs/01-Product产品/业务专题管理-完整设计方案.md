```javascript
/**
 * 验证单条议题数据
 * @returns { isValid: boolean, errors: string[], warnings: string[] }
 */
function validateIssueRow(row, sourceSystem) {
  const errors = [];
  const warnings = [];
  
  // 必填字段校验
  const required = ['会议名称', '会议日期', '议题编号', '议题标题', '议题类型', '主责部门', '提案人', '状态'];
  required.forEach(field => {
    if (!row[field] || String(row[field]).trim() === '') {
      errors.push(`缺少必填字段: ${field}`);
    }
  });
  
  // 议题编号格式校验
  if (row['议题编号'] && !safeIssueId(row['议题编号'])) {
    errors.push(`议题编号格式错误: ${row['议题编号']}`);
  }
  
  // 日期格式校验
  if (row['会议日期'] && !isValidDate(row['会议日期'])) {
    errors.push(`日期格式错误: ${row['会议日期']}`);
  }
  
  // CSV 公式注入检测
  const textFields = ['议题标题', '议题内容', '决议内容', '行动项'];
  textFields.forEach(field => {
    if (row[field] && hasCsvFormulaInjection(row[field])) {
      warnings.push(`字段 "${field}" 可能包含 CSV 公式注入，已自动清理`);
      row[field] = sanitizeCsvCell(row[field]);
    }
  });
  
  return { isValid: errors.length === 0, errors, warnings, cleanedRow: row };
}

/**
 * CSV 公式注入检测
 * 检测 =, +, -, @, \t, \r 等前缀
 */
function hasCsvFormulaInjection(text) {
  if (typeof text !== 'string') return false;
  const dangerousPrefixes = ['=', '+', '-', '@', '\t', '\r'];
  const trimmed = text.trim();
  return dangerousPrefixes.some(p => trimmed.startsWith(p));
}

/**
 * 清理 CSV 单元格，防止公式注入
 */
function sanitizeCsvCell(text) {
  if (typeof text !== 'string') return text;
  const trimmed = text.trim();
  const dangerousPrefixes = ['=', '+', '-', '@', '\t', '\r'];
  if (dangerousPrefixes.some(p => trimmed.startsWith(p))) {
    // 前缀加单引号，Excel 会将其视为文本而非公式
    return "'" + trimmed;
  }
  return text;
}

/**
 * 议题 ID 安全校验
 */
function safeIssueId(id) {
  return typeof id === 'string' && /^(ST|AT)-[0-9]{4}-Q[1-4]-[0-9]+$/.test(id);
}
```

```javascript
const STORAGE_LIMIT = 5 * 1024 * 1024; // 5MB
const STORAGE_WARNING_THRESHOLD = 0.8;  // 80% 预警

function checkStorageCapacity() {
  let used = 0;
  for (let key in localStorage) {
    if (localStorage.hasOwnProperty(key)) {
      used += (localStorage.getItem(key) || '').length * 2; // UTF-16 ≈ 2 bytes/char
    }
  }
  const ratio = used / STORAGE_LIMIT;
  return {
    usedBytes: used,
    limitBytes: STORAGE_LIMIT,
    ratio,
    isWarning: ratio > STORAGE_WARNING_THRESHOLD,
    isCritical: ratio > 0.95
  };
}
```

```javascript
function saveTopicsWithVersion(topics) {
  const currentRaw = localStorage.getItem(STORAGE_KEYS.TOPICS);
  const currentHash = currentRaw ? simpleHash(currentRaw) : '';
  const lastSavedHash = window._lastSavedHash || '';
  
  if (lastSavedHash && currentHash !== lastSavedHash) {
    // 数据已被其他标签页修改
    if (!confirm('检测到其他页面已修改数据，是否覆盖？建议先刷新页面查看最新数据。')) {
      return false;
    }
  }
  
  localStorage.setItem(STORAGE_KEYS.TOPICS, JSON.stringify(topics));
  window._lastSavedHash = simpleHash(JSON.stringify(topics));
  return true;
}

function simpleHash(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    const char = str.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  return hash.toString(16);
}
```

### 4.3 AI 全局分析报告（v2.1 增强：算法优化 + 缓存）

#### 4.3.1 v2.1 改进：聚类算法优化

原 v2.0 方案使用 O(n²) 双重循环。v2.1 改为**预构建索引 + 分组比较**，平均复杂度降为 O(n)。

```javascript
const AnalysisRules = {
  /**
   * v2.1 优化版：议题聚类（共享 KPI + 时间重叠 + 部门关联）
   * 复杂度：O(n) 平均（通过 KPI 索引预分组）
   */
  findClusters(issues) {
    // Step 1: 构建 KPI → 议题索引（O(n)）
    const kpiIndex = new Map(); // kpiId -> Set(issueId)
    issues.forEach(issue => {
      (issue.relatedKpis || []).forEach(kpiId => {
        if (!kpiIndex.has(kpiId)) kpiIndex.set(kpiId, new Set());
        kpiIndex.get(kpiId).add(issue.issueId);
      });
    });
    
    // Step 2: 找出共享 KPI 的议题对（O(共享KPI的议题数))）
    const candidatePairs = new Map(); // "id1|id2" -> { sharedKpis: [], evidence: {} }
    kpiIndex.forEach((issueSet, kpiId) => {
      const issueList = Array.from(issueSet);
      if (issueList.length < 2) return;
      for (let i = 0; i < issueList.length; i++) {
        for (let j = i + 1; j < issueList.length; j++) {
          const key = [issueList[i], issueList[j]].sort().join('|');
          if (!candidatePairs.has(key)) {
            candidatePairs.set(key, { sharedKpis: [], evidence: {} });
          }
          candidatePairs.get(key).sharedKpis.push(kpiId);
        }
      }
    });
    
    // Step 3: 对候选对进行完整评估（仅评估有共享 KPI 的对，大幅缩小比较范围）
    const clusters = [];
    candidatePairs.forEach((data, key) => {
      const [id1, id2] = key.split('|');
      const issue1 = issues.find(i => i.issueId === id1);
      const issue2 = issues.find(i => i.issueId === id2);
      if (!issue1 || !issue2) return;
      
      const timeOverlap = checkTimeOverlap(issue1, issue2);
      const deptRelated = checkDeptRelation(issue1, issue2);
      
      // v2.1: 增加置信度阈值过滤，只保留高置信度的聚类
      const confidence = calculateConfidence(data.sharedKpis.length, timeOverlap, deptRelated);
      if (confidence < 0.6) return; // 过滤低置信度
      
      clusters.push({
        issues: [issue1, issue2],
        evidence: { sharedKpis: data.sharedKpis, timeOverlap, deptRelated },
        confidence
      });
    });
    
    return mergeClusters(clusters);
  },
  
  // 其余规则保持不变...
};
```

```javascript
/**
 * 生成全局报告（带缓存）
 */
function generateGlobalReport(issues, reportType) {
  // 计算议题数据校验和
  const checksum = simpleHash(JSON.stringify(issues.map(i => i.issueId).sort()));
  
  // 检查缓存
  const cached = loadCachedReport(reportType, checksum);
  if (cached) {
    return { ...cached, isCached: true };
  }
  
  // 生成新报告
  const report = doGenerateReport(issues, reportType);
  report.dataChecksum = checksum;
  report.isCached = false;
  
  // 保存缓存
  saveCachedReport(reportType, checksum, report);
  return report;
}

function loadCachedReport(reportType, checksum) {
  try {
    const raw = localStorage.getItem(`${STORAGE_KEYS.REPORTS}_${reportType}`);
    if (!raw) return null;
    const cache = JSON.parse(raw);
    return cache.checksum === checksum ? cache.report : null;
  } catch {
    return null;
  }
}

function saveCachedReport(reportType, checksum, report) {
  try {
    localStorage.setItem(`${STORAGE_KEYS.REPORTS}_${reportType}`, JSON.stringify({
      checksum,
      report,
      cachedAt: new Date().toISOString()
    }));
  } catch (e) {
    // 缓存失败不影响主流程
    console.warn('[ReportCache] 缓存保存失败:', e.message);
  }
}
```

```javascript
/**
 * 渲染简化版知识地图（SVG）
 * 节点：专题（中心）、议题（环绕）、KPI（底部）
 * 边：直接驱动（实线）、相关支撑（虚线）、KPI 关联（点线）
 */
function renderKnowledgeMap(containerId, topic, linkedIssues, linkedKpis) {
  const container = document.getElementById(containerId);
  if (!container) return;
  
  const width = 600;
  const height = 400;
  const centerX = width / 2;
  const centerY = height / 2 - 30;
  
  // 节点布局计算
  const topicNode = { x: centerX, y: centerY, type: 'topic', label: topic.name };
  const issueNodes = linkedIssues.map((li, i) => {
    const angle = (2 * Math.PI * i) / linkedIssues.length - Math.PI / 2;
    const radius = 120;
    return {
      x: centerX + radius * Math.cos(angle),
      y: centerY + radius * Math.sin(angle),
      type: li.sourceSystem,
      label: li.issueId,
      relationType: li.relationType
    };
  });
  
  // 构建 SVG
  let svg = `<svg viewBox="0 0 ${width} ${height}" style="width:100%;height:auto;">`;
  
  // 绘制边
  issueNodes.forEach(node => {
    const isDirect = node.relationType === 'direct';
    const strokeDash = isDirect ? 'none' : '5,5';
    svg += `<line x1="${topicNode.x}" y1="${topicNode.y}" x2="${node.x}" y2="${node.y}" 
                   stroke="#6366f1" stroke-width="${isDirect ? 2 : 1}" 
                   stroke-dasharray="${strokeDash}" opacity="0.6"/>`;
  });
  
  // 绘制议题节点
  issueNodes.forEach(node => {
    const color = node.type === 'ST' ? '#8b5cf6' : '#06b6d4';
    svg += `<circle cx="${node.x}" cy="${node.y}" r="20" fill="${color}" opacity="0.9"/>
            <text x="${node.x}" y="${node.y + 4}" text-anchor="middle" fill="white" font-size="10">${node.type}</text>
            <text x="${node.x}" y="${node.y + 35}" text-anchor="middle" fill="#374151" font-size="10">${escapeHtml(node.label)}</text>`;
  });
```

### 8.3 v2.1 安全函数完整实现

```javascript
/**
 * 增强版 HTML 转义 — v2.1 覆盖所有文本字段
 */
function escapeHtml(text) {
  if (text == null) return '';
  const div = document.createElement('div');
  div.textContent = String(text);
  return div.innerHTML;
}

/**
 * 议题 ID 安全校验 — v2.1 新增
 */
function safeIssueId(id) {
  return typeof id === 'string' && /^(ST|AT)-[0-9]{4}-Q[1-4]-[0-9]+$/.test(id);
}

/**
 * CSV 公式注入检测 — v2.1 新增
 */
function hasCsvFormulaInjection(text) {
  if (typeof text !== 'string') return false;
  const trimmed = text.trim();
  const dangerous = ['=', '+', '-', '@', '\t', '\r'];
  return dangerous.some(p => trimmed.startsWith(p));
}

/**
 * CSV 单元格清理 — v2.1 新增
 * 在危险字符前加单引号，使 Excel/Google Sheets 将其视为文本
 */
function sanitizeCsvCell(text) {
  if (typeof text !== 'string') return text;
  const trimmed = text.trim();
  const dangerous = ['=', '+', '-', '@', '\t', '\r'];
  if (dangerous.some(p => trimmed.startsWith(p))) {
    return "'" + trimmed;
  }
  return text;
}
```
